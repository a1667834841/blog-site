" ") -}} {{- end -}} {{- else -}} {{- range site.Params.SocialIcons -}} {{- $sameAs = $sameAs | append (trim .url " ") -}} {{- end -}} {{- end -}} {{- $common := dict "@context" "https://schema.org" "@type" $publisherType "name" site.Title "url" site.Home.Permalink "description" (site.Params.description | plainify | truncate 180) "sameAs" $sameAs -}} {{- $imgURL := (site.Params.assets.favicon | default "favicon.ico" | absURL) -}} {{- if (eq site.Params.schema.publisherType "Person") -}} {{- $common = merge $common (dict "image" $imgURL) -}} {{- else -}} {{- $common = merge $common (dict "logo" $imgURL) -}} {{- end -}} {{- else if (or .IsPage .IsSection) -}} {{/* BreadcrumbList */}} {{- $url := replace .Parent.Permalink ( printf "%s" site.Home.Permalink) "" -}} {{- $langURL :=
